I pass the name of an existing file, the new name for the copies and the numer

./scriptname filename.extension newfilename.extension number_of_copies

#!/bin/bash

x=0
extension=$(echo $1 | sed -n -e 's/[[:print:]]*\.//gp')
if [ -z $extension ]
then

     while [ $x -lt $3 ]
     do
         cp $1 $2$x
         x=`expr $x + 1`
     done

else

     while [ $x -lt $3 ]
     do
         cp $1 $2$x.$extension
         x=`expr $x + 1`
     done
fi
