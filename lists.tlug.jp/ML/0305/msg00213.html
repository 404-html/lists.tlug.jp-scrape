Hoping this works same as your script.

-----------------
#!/bin/bash
extension=$(echo $1 | sed -n -e 's/[[:print:]]*\.//gp')
extension=${extension:+"."$extension}

for i in `seq 0 $(($3 - 1))`;do
	cp $1 $2$i$extension
done
