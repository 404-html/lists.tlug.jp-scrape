Uvator,

There is a readlink(2) system call on HP/UX.

So.....

#include <unistd.h>
#include <limits.h>
#include <stdio.h>
 
int main(int argc, char** argv) {
        char buf[PATH_MAX + 1];
 
        if ( (argc < 2) || (!argv[1]) ) {
                fprintf(stderr, "%s: usage: <symlink>\n", argv[0]);
                return 1;
        }
 
        if ( readlink(argv[1], buf, PATH_MAX) == -1) {
                perror("readlink");
                return 1;
        }
 
        printf("%s\n", buf);
 
        return 0;
} 

I can compile this for you on HP/UX 10.2 if you like.
