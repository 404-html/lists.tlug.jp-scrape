I just looked it up in an old paper, try this:

\documentclass[12pt]{article}

\usepackage{amsmath,amscd,amssymb,subfigure,epsfig,array}
\twocolumn

\begin{document}
\title{Spectral Properties of the Chalker-Coddington Network}
\author{Marcus Metzler}
\author{\small\parbox{14.1cm}{\small
%\begin{abstract}
We numerically investigate the spectral statistics of pseudo-energies 
for the unitary network operator $U$ of the Chalker--Coddington network. 
We show that those statistics agree with results obtained for other models
of quantum Hall systems, by analyzing the shape of the spacing distribution
$P(s)$. 
%\end{abstract}
PACS: 71.23.AN, 71.30.+h, 73.40.Hm, 71.50.+t, 72.15.Rn
}}
\maketitle
\section{Introduction}
\label{sec:intro}
