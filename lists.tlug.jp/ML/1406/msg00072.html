Ah, the joys of libtool.

$ cd /tmp
$ tar -zxf /path/to/sleuthkit-4.1.3
$ cd sleuthkit-4.1.3
$ ./configure
$ find . -name Makefile -exec sed -i "/^LDFLAGS /s/-static/-all-static/" {} \+
$ make
$ find tools -type f -executable -exec file {} \;
