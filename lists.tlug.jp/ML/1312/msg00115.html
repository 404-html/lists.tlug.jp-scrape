The following produces the same on Debian and Fedora:

        echo -e "$l\t $(LC_ALL=ja_JP.utf8 date +%A)"; 
        LC_ALL=ja_JP.$l date +%A|xxd; echo; 
done | md5sum -c <(echo "31f0f83e7fe3dacb7b288c101cd1debd  -")
