Is this what you need:
$ cd /tmp
$ mkdir -p d1/D{2,3,4,5,6} d1/G{3,4}/F5
$ touch d1/D{2,3,4,5,6}/some_file d1/G{3,4}/F5/other d1/
$ touch d1/D{2,3,4,5,6}/some_file d1/G{3,4}/F5/other d1/D{2,3,4,5,6}/evenMORE
$ LC_ALL=C tree d1/
d1/

|   `-- some_file
|-- D3

|-- D4

|-- G3
