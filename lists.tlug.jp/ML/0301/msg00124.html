       for I in *.??3
       do
           mpg123 -s "$I" | cdrecord dev=0,4,0 speed=2 -v -audio -swab -nofix -
       done
       cdrecord speed=2 dev=0,4,0 -v -fix
       cdrecord dev=0,4,0 -eject
