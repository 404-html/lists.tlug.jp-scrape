when I "su"):

br@example.com:~$ cat ~br-test/.bashrc
export FOO=BAR
br@example.com:~$ echo $FOO

br@example.com:~$ su - br-test
Password:
br-test@example.com:~$ echo $FOO
BAR
br-test@example.com:~$ logout
br@example.com:~$ cat ~br-test/.bashrc
export FOO=BAR
br@example.com:~$ echo $FOO

br@example.com:~$ su br-test
Password:
br-test@example.com:/home/br$ echo $FOO
BAR
