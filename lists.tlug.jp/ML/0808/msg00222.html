Sure. Here's mine:

: jglover@example.com; cat .config/sakura/sakura.conf

[sakura]
forecolor=#00ff00
backcolor=#000000
boldcolor=#00ff00
opacity_level=80
fake_transparency=No
background=none
font=Bitstream Vera Sans Mono 7
show_always_first_tab=Yes
word_chars=-A-Za-z0-9,./?%&#_~:
add_tab_accelerator=4
del_tab_accelerator=5
switch_tab_accelerator=4
copy_accelerator=5
add_tab_key=116
del_tab_key=87
copy_key=67
paste_key=86
prev_tab_key=65365
next_tab_key=65366
scrollbar=false
scrollbar_accelerator=5
scrollbar_key=83
fullscreen_key=65480
